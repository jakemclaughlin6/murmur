    ```dart
    import 'package:flutter/material.dart';
    import 'package:flutter_riverpod/flutter_riverpod.dart';
    import 'package:go_router/go_router.dart';
    import '../../core/db/app_database.dart';
    import 'book_card.dart';
    import 'book_card_shimmer.dart';
    import 'import_service.dart';

    class LibraryGrid extends ConsumerWidget {
      final List&lt;Book&gt; books;
      final ValueChanged&lt;int&gt; onLongPress;
      const LibraryGrid({
        super.key,
        required this.books,
        required this.onLongPress,
      });

      int _columnCount(BuildContext context) {
        final size = MediaQuery.sizeOf(context);
        final shortest = size.shortestSide;
        final isLandscape = size.width &gt; size.height;
        if (shortest &lt; 600) {
          return isLandscape ? 3 : 2; // phone
        } else {
          return isLandscape ? 6 : 4; // tablet
        }
      }

      @override
      Widget build(BuildContext context, WidgetRef ref) {
        final importState = ref.watch(importNotifierProvider);
        final parsing = importState.whereType&lt;ImportParsing&gt;().toList();
        final columnCount = _columnCount(context);

        // Prepend shimmer cards for in-flight imports per D-11
        final items = &lt;Widget&gt;[
          for (final p in parsing) BookCardShimmer(filename: p.filename),
          for (final book in books)
            BookCard(
              book: book,
              onTap: () =&gt; context.push('/reader/${book.id}'),
              onLongPress: () =&gt; onLongPress(book.id),
            ),
        ];

        return SliverPadding(
          padding: const EdgeInsets.all(12),
          sliver: SliverGrid(
            gridDelegate: SliverGridDelegateWithFixedCrossAxisCount(
              crossAxisCount: columnCount,
              crossAxisSpacing: 12,
              mainAxisSpacing: 16,
              childAspectRatio: 0.55, // cover-dominant aspect
            ),
            delegate: SliverChildBuilderDelegate(
              (context, index) =&gt; items[index],
              childCount: items.length,
            ),
          ),
        );
      }
    }
    ```

    ```dart
    import 'dart:async';
    import 'package:flutter/material.dart';
    import 'package:flutter_riverpod/flutter_riverpod.dart';
    import '../../core/theme/clay_colors.dart';
    import 'library_provider.dart';

    class LibrarySearchBar extends ConsumerStatefulWidget {
      const LibrarySearchBar({super.key});

      @override
      ConsumerState&lt;LibrarySearchBar&gt; createState() =&gt; _LibrarySearchBarState();
    }

    class _LibrarySearchBarState extends ConsumerState&lt;LibrarySearchBar&gt; {
      final _controller = TextEditingController();
      Timer? _debounce;

      @override
      void dispose() {
        _controller.dispose();
        _debounce?.cancel();
        super.dispose();
      }

      void _onChanged(String value) {
        _debounce?.cancel();
        _debounce = Timer(const Duration(milliseconds: 300), () {
          ref.read(libraryNotifierProvider.notifier).setSearchQuery(value);
        });
      }

      @override
      Widget build(BuildContext context) {
        return Padding(
          padding: const EdgeInsets.symmetric(horizontal: 12, vertical: 8),
          child: TextField(
            controller: _controller,
            onChanged: _onChanged,
            decoration: InputDecoration(
              hintText: 'Search title or author',
              prefixIcon: const Icon(Icons.search, color: ClayColors.textTertiary),
              suffixIcon: _controller.text.isEmpty
                  ? null
                  : IconButton(
                      icon: const Icon(Icons.clear),
                      onPressed: () {
                        _controller.clear();
                        _onChanged('');
                      },
                    ),
              filled: true,
              fillColor: ClayColors.surface,
              border: OutlineInputBorder(
                borderRadius: BorderRadius.circular(24),
                borderSide: const BorderSide(color: ClayColors.borderSubtle),
              ),
            ),
          ),
        );
      }
    }
    ```

    ```dart
    Future&lt;void&gt; showBookContextSheet(BuildContext context, Book book) {
      return showModalBottomSheet(

        backgroundColor: ClayColors.surface,
        builder: (ctx) =&gt; _BookContextSheet(book: book),
      );

    class _BookContextSheet extends ConsumerWidget {
      final Book book;
      const _BookContextSheet({required this.book});

      @override
      Widget build(BuildContext context, WidgetRef ref) {
        return SafeArea(
          child: Column(
            mainAxisSize: MainAxisSize.min,
            children: [
              ListTile(
                leading: const Icon(Icons.info_outline),
                title: const Text('Book Info'),
                onTap: () async {
                  Navigator.of(context).pop();
                  await _showInfoDialog(context, ref, book);
                },
              ),
              ListTile(
                leading: const Icon(Icons.delete_outline, color: Colors.red),
                title: const Text('Delete', style: TextStyle(color: Colors.red)),
                onTap: () async {
                  Navigator.of(context).pop();
                  final confirmed = await showDialog&lt;bool&gt;(
                    context: context,
                    builder: (_) =&gt; AlertDialog(
                      title: Text('Delete "${book.title}"?'),
                      content: const Text('This will remove the book and its reading progress.'),
                      actions: [
                        TextButton(onPressed: () =&gt; Navigator.pop(context, false), child: const Text('Cancel')),
                        TextButton(onPressed: () =&gt; Navigator.pop(context, true), child: const Text('Delete')),
                      ],
                    ),
                  );
                  if (confirmed == true) {
                    await ref.read(libraryNotifierProvider.notifier).deleteBook(book.id);
                  }
                },
              ),
            ],
          ),
        );
      }

      Future&lt;void&gt; _showInfoDialog(BuildContext context, WidgetRef ref, Book book) async {
        final db = ref.read(appDatabaseProvider);
        final chapterCount = await (db.selectOnly(db.chapters)
              ..addColumns([db.chapters.id.count()])
              ..where(db.chapters.bookId.equals(book.id)))
            .getSingle()
            .then((row) =&gt; row.read(db.chapters.id.count()) ?? 0);
        var fileSize = 0;
        final f = File(book.filePath);
        if (f.existsSync()) fileSize = f.lengthSync();
        if (!context.mounted) return;
        await showDialog(
          context: context,
          builder: (_) =&gt; AlertDialog(
            title: Text(book.title),
            content: Column(
              mainAxisSize: MainAxisSize.min,
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                Text('Author: ${book.author ?? "Unknown"}'),
                Text('Chapters: $chapterCount'),
                Text('Size: ${(fileSize / 1024 / 1024).toStringAsFixed(1)} MB'),
                Text('Imported: ${book.importDate.toLocal().toString().split(' ').first}'),
              ],
            ),
            actions: [TextButton(onPressed: () =&gt; Navigator.pop(context), child: const Text('Close'))],
          ),
        );
      }
    ```

    ```dart
    import 'package:flutter/material.dart';
    import 'package:flutter_riverpod/flutter_riverpod.dart';
    import '../../core/db/app_database.dart';
    import '../../core/db/app_database_provider.dart';
    import '../../core/theme/clay_colors.dart';
    import 'book_context_sheet.dart';

    import 'library_search_bar.dart';

    class LibraryScreen extends ConsumerWidget {
      const LibraryScreen({super.key});

      @override
      Widget build(BuildContext context, WidgetRef ref) {
        // Listen for import failures and show snackbars per D-12
        ref.listen&lt;List&lt;ImportState&gt;&gt;(importNotifierProvider, (prev, next) {
          final prevFailed = (prev ?? []).whereType&lt;ImportFailed&gt;().toSet();
          final nextFailed = next.whereType&lt;ImportFailed&gt;().toSet();
          for (final f in nextFailed.difference(prevFailed)) {
            ScaffoldMessenger.of(context).showSnackBar(
              SnackBar(content: Text('Could not import ${f.filename} — ${f.reason}')),
            );
          }
        });

        final libAsync = ref.watch(libraryNotifierProvider);
        return Scaffold(
          backgroundColor: ClayColors.background,
          body: libAsync.when(
            loading: () =&gt; const Center(child: CircularProgressIndicator()),
            error: (e, _) =&gt; Center(child: Text('Error: $e')),
            data: (state) {
              final hasBooks = state.books.isNotEmpty;
              final hasQuery = state.searchQuery.isNotEmpty;
              final importParsing = ref.watch(importNotifierProvider).whereType&lt;ImportParsing&gt;().isNotEmpty;

              // Empty first-import state: no books, no query, no imports in flight
              if (!hasBooks &amp;&amp; !hasQuery &amp;&amp; !importParsing) {
                return _EmptyFirstImport(
                  onImport: () =&gt; ref.read(importNotifierProvider.notifier).pickAndImport(),
                );
              }
              return CustomScrollView(
                slivers: [
                  SliverAppBar(
                    title: const Text('Library'),
                    pinned: true,
                    actions: [
                      IconButton(
                        icon: const Icon(Icons.add),
                        onPressed: () =&gt; ref.read(importNotifierProvider.notifier).pickAndImport(),
                      ),
                    ],
                  ),
                  const SliverToBoxAdapter(child: LibrarySearchBar()),
                  const SliverToBoxAdapter(child: LibrarySortChips()),
                  if (!hasBooks &amp;&amp; hasQuery)
                    const SliverFillRemaining(
                      hasScrollBody: false,
                      child: Center(child: Text('No books match your search')),
                    )
                  else
                    LibraryGrid(
                      books: state.books,
                      onLongPress: (bookId) async {
                        final book = state.books.firstWhere((b) =&gt; b.id == bookId);
                        await showBookContextSheet(context, book);
                      },
                    ),
                ],
              );
            },
          ),
        );
      }
    }
    ```
