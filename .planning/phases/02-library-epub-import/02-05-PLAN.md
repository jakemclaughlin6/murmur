    ```dart
    import 'dart:io';
    import 'package:file_picker/file_picker.dart';
    import 'package:path/path.dart' as p;
    import 'package:path_provider/path_provider.dart';
    import 'package:riverpod_annotation/riverpod_annotation.dart';
    import '../../core/db/app_database.dart';
    import '../../core/db/app_database_provider.dart';
    import '../../core/epub/epub_parser.dart';
    import '../../core/epub/epub_parser_isolate.dart';
    import '../../core/epub/drm_detector.dart';
    import '../../core/epub/parse_result.dart';
    import '../../core/epub/block_json.dart';

    part 'import_service.g.dart';

    sealed class ImportState { ... }
    class ImportParsing extends ImportState { ... }
    class ImportSuccess extends ImportState { final int bookId; ... }
    class ImportFailed extends ImportState { final String reason; ... }

    @Riverpod(keepAlive: true)
    class ImportNotifier extends _$ImportNotifier {
      @override
      List&lt;ImportState&gt; build() =&gt; const [];

      Future&lt;void&gt; pickAndImport() async {
        final result = await FilePicker.platform.pickFiles(
          allowMultiple: true,
          type: FileType.custom,
          allowedExtensions: const ['epub'],
        );
        if (result == null || result.files.isEmpty) return;
        final paths = result.files.map((f) =&gt; f.path!).where((p) =&gt; p.isNotEmpty).toList();
        await importFromPaths(paths);
      }

      Future&lt;void&gt; importFromPaths(List&lt;String&gt; paths) async {
        // Emit ImportParsing for each path optimistically (D-11).
        state = [for (final path in paths) ImportParsing(p.basename(path))];

        final db = ref.read(appDatabaseProvider);
        final appDocsDir = await getApplicationDocumentsDirectory();
        final coversDir = Directory(p.join(appDocsDir.path, 'covers'));
        if (!coversDir.existsSync()) coversDir.createSync(recursive: true);

        final newStates = [...state];
        for (var i = 0; i &lt; paths.length; i++) {
          final path = paths[i];
          final filename = p.basename(path);
          try {
            // 1. Read bytes
            final bytes = await File(path).readAsBytes();
            // 2. Parse in isolate
            final parsed = await parseEpubInIsolate(bytes)
                .timeout(const Duration(seconds: 30));
            // 3. Copy EPUB to app documents dir
            final destPath = p.join(appDocsDir.path, 'books', filename);
            Directory(p.dirname(destPath)).createSync(recursive: true);
            await File(path).copy(destPath);
            // 4. Insert Books row (cover path filled in after)
            final bookId = await db.into(db.books).insert(
              BooksCompanion.insert(
                title: parsed.title,
                author: Value(parsed.author),
                filePath: destPath,
                coverPath: const Value(null),
                importDate: DateTime.now(),
              ),
            );
            // 5. Write cover if present, respecting 10 MB cap
            String? coverPath;
            if (parsed.coverBytes != null) {
              if (parsed.coverBytes!.length &gt; 10 * 1024 * 1024) {
                // Cover too large — skip cover but keep book
                // or fail the whole import per T-02-04-05. Policy: skip cover only.
              } else {
                coverPath = p.join(coversDir.path, '$bookId.jpg');
                await File(coverPath).writeAsBytes(parsed.coverBytes!);
                await (db.update(db.books)..where((b) =&gt; b.id.equals(bookId)))
                    .write(BooksCompanion(coverPath: Value(coverPath)));
              }
            }
            // 6. Insert chapters with blocks_json
            for (final chapter in parsed.chapters) {
              await db.into(db.chapters).insert(
                ChaptersCompanion.insert(
                  bookId: bookId,
                  orderIndex: chapter.orderIndex,
                  title: Value(chapter.title),
                  blocksJson: blocksToJsonString(chapter.blocks),
                ),
              );
            }
            newStates[i] = ImportSuccess(filename, bookId);
          } on DrmDetectedException {
            newStates[i] = ImportFailed(filename, 'DRM-protected');
          } on EpubParseException catch (e) {
            newStates[i] = ImportFailed(filename, 'Corrupt: ${e.reason}');
          } on SqliteException catch (e) {
            // Unique constraint violation -&gt; already imported
            if (e.message.contains('UNIQUE')) {
              newStates[i] = ImportFailed(filename, 'Already in library');
            } else {
              newStates[i] = ImportFailed(filename, 'Database error');
            }
          } catch (e) {
            newStates[i] = ImportFailed(filename, 'Unknown error');
          }
          state = List.unmodifiable(newStates);
        }
      }
    }
    ```

    Notes:
    - `@Riverpod(keepAlive: true)` because the import service outlives widget rebuilds.
    - State is `List&lt;ImportState&gt;` so Plan 07's library grid can overlay shimmer cards per-book.
    - The 30s timeout on parse protects against T-02-04-01 (zip bomb / stall DoS).
    - The 10 MB cover cap enforces T-02-04-05.
    - Path is validated inside the app documents dir indirectly: we `File(path).copy(destPath)` where `destPath` is OUR construction — the user's source path never becomes the stored `file_path`.
    - Per D-14: `importFromPaths` is the single pipeline both file_picker (Task 2) and Share intent (Task 3) call.

    ```dart
    import 'dart:async';
    import 'package:riverpod_annotation/riverpod_annotation.dart';
    import 'package:receive_sharing_intent_plus/receive_sharing_intent_plus.dart';
    import 'import_service.dart';

    part 'share_intent_listener.g.dart';

    @Riverpod(keepAlive: true)
    class ShareIntentListener extends _$ShareIntentListener {
      StreamSubscription&lt;List&lt;SharedMediaFile&gt;&gt;? _subscription;

      @override
      Future&lt;void&gt; build() async {
        // 1. Handle initial launch intent (app opened from Share while killed)
        final initial = await ReceiveSharingIntent.instance.getInitialMedia();
        if (initial.isNotEmpty) {
          _route(initial);
          await ReceiveSharingIntent.instance.reset();
        }
        // 2. Listen for runtime shares (app already running)
        _subscription = ReceiveSharingIntent.instance
            .getMediaStream()
            .listen(_route);
        ref.onDispose(() =&gt; _subscription?.cancel());
      }

      void _route(List&lt;SharedMediaFile&gt; files) {
        final epubPaths = files
            .where((f) =&gt; f.path.toLowerCase().endsWith('.epub'))
            .map((f) =&gt; f.path)
            .toList();
        if (epubPaths.isEmpty) return;
        ref.read(importNotifierProvider.notifier).importFromPaths(epubPaths);
      }
    }
    ```

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-05-01 | Spoofing | Share intent from malicious app sending non-EPUB with EPUB MIME | mitigate | ShareIntentListener filters by `.epub` extension AFTER the OS MIME check. Parser then validates zip structure and OPF manifest — non-EPUB bytes fail at parseEpub and emit ImportFailed. |
| T-02-05-02 | Tampering | Path traversal via source file_path | mitigate | ImportService uses `p.basename(sourcePath)` as the destination filename and always constructs destPath from `appDocsDir.path` + 'books/' + basename. User-supplied path NEVER becomes the stored `file_path`. |
