    **Fill in `test/library/persistence_test.dart`** (unskip Plan 02 stub):

    ```dart
    import 'dart:io';
    import 'package:drift/native.dart';
    import 'package:flutter_riverpod/flutter_riverpod.dart';
    import 'package:flutter_test/flutter_test.dart';
    import 'package:murmur/core/db/app_database.dart';
    import 'package:murmur/core/db/app_database_provider.dart';
    import 'package:murmur/features/library/import_service.dart';

    void main() {
      test('book survives DB reopen (LIB-11)', () async {
        // Use a persistent file so reopen actually tests persistence,
        // not just in-memory state.
        final tempDir = Directory.systemTemp.createTempSync('murmur_persist_');
        final dbFile = File('${tempDir.path}/test.sqlite');
        addTearDown(() =&gt; tempDir.deleteSync(recursive: true));

        // 1. First session: open DB, import minimal.epub
        var db1 = AppDatabase(NativeDatabase(dbFile));
        var container1 = ProviderContainer(overrides: [
          appDatabaseProvider.overrideWithValue(db1),
        ]);
        addTearDown(container1.dispose);

        final notifier = container1.read(importNotifierProvider.notifier);
        await notifier.importFromPaths([
          'test/fixtures/epub/minimal.epub',
        ]);

        // Wait for import to complete
        var states = container1.read(importNotifierProvider);
        expect(states.whereType&lt;ImportSuccess&gt;(), isNotEmpty, reason: 'Import must succeed');

        final booksBefore = await db1.select(db1.books).get();
        expect(booksBefore, hasLength(1));
        final titleBefore = booksBefore.first.title;

        // 2. Close first DB + container
        await db1.close();
        container1.dispose();

        // 3. Second session: reopen the same file
        var db2 = AppDatabase(NativeDatabase(dbFile));
        addTearDown(() async =&gt; db2.close());

        final booksAfter = await db2.select(db2.books).get();
        expect(booksAfter, hasLength(1));
        expect(booksAfter.first.title, titleBefore);

        // Verify chapters also persisted (Plan 03 schema)
        final chaptersAfter = await db2.select(db2.chapters).get();
        expect(chaptersAfter, isNotEmpty, reason: 'Chapters must persist with the book');
      });
    }
    ```
