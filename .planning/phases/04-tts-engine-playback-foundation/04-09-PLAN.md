       ```dart
       // Usage: dart run tool/measure_tts_latency.dart build/integration_response_data/phase_04_play_tap.timeline.json
       void main(List<String> args) {
         final json = jsonDecode(File(args[0]).readAsStringSync()) as Map<String, dynamic>;
         final events = (json['traceEvents'] as List).cast<Map<String, dynamic>>();
         final playTapBegin = events.firstWhere((e) => e['name'] == 'tts.play_tap' &amp;&amp; e['ph'] == 'B');
         final setFileEnd   = events.lastWhere((e)  => e['name'] == 'tts.set_file_play' &amp;&amp; e['ph'] == 'E');
         final latencyUs    = (setFileEnd['ts'] as int) - (playTapBegin['ts'] as int);
         final latencyMs    = latencyUs / 1000;
         stdout.writeln('TTS play-tap → first-audio-frame latency: ${latencyMs.toStringAsFixed(1)}ms');
         exit(latencyMs < 300 ? 0 : 1);   // matches TTS-10 threshold
       }
       ```
