       test('package:http is imported ONLY by lib/features/tts/model/model_downloader.dart', () async {
         final violations = <String>[];
         await for (final e in Directory('lib').list(recursive: true)) {
           if (e is File &amp;&amp; e.path.endsWith('.dart') &amp;&amp; !e.path.endsWith('.g.dart')) {
             final src = await e.readAsString();
             if (RegExp(r"""import\s+['"]package:http/""").hasMatch(src)) {
               if (!e.path.endsWith('/model_downloader.dart')) violations.add(e.path);
             }
           }
         }
         expect(violations, isEmpty);
       });

       test('package:sherpa_onnx is imported ONLY by lib/features/tts/isolate/sherpa_tts_engine.dart (+ debug spike)', () async {
         final allowed = {'lib/features/tts/isolate/sherpa_tts_engine.dart', 'lib/features/tts/spike/spike_page.dart'};
         final violations = <String>[];
         await for (final e in Directory('lib').list(recursive: true)) {
           if (e is File &amp;&amp; e.path.endsWith('.dart') &amp;&amp; !e.path.endsWith('.g.dart')) {
             final src = await e.readAsString();
             if (RegExp(r"""import\s+['"]package:sherpa_onnx/""").hasMatch(src)) {
               if (!allowed.contains(e.path)) violations.add(e.path);
             }
           }
         }
         expect(violations, isEmpty);
       });

       test('no analytics/telemetry/crashlytics packages anywhere', () async {
         final banned = ['firebase', 'sentry', 'crashlytics', 'amplitude', 'mixpanel', 'segment'];
         final violations = <String>[];
         await for (final e in Directory('lib').list(recursive: true)) {
           if (e is File &amp;&amp; e.path.endsWith('.dart')) {
             final src = (await e.readAsString()).toLowerCase();
             for (final b in banned) {
               if (src.contains('package:$b')) violations.add('${e.path} imports $b');
             }
           }
         }
         expect(violations, isEmpty);
       });
